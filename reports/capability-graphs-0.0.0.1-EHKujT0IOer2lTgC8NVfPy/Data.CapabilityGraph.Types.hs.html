   26 -- | The primary graph structure, most likely will only be interacted w/ wrapped in a more featureful level
   27 newtype Graph v = GraphT
   28   { -- | underlying data structure, representing the graph as a hashmap from an INT -> a tuple containing a node (value)
   29     -- | a hashset of ints (w/ the ints being the keys of the nodes parent)
   30     unHashMap :: HM.HashMap Int (Node v, HS.HashSet Int)
   31   }
   32 
   33 instance Functor Graph where
   34   f `fmap` (GraphT hm) = GraphT $ HM.map (\(NodeT k n, hs) -> (NodeT k (f n), hs)) hm
   35 
   36 data Node v = NodeT {key :: Int, unNode :: v}
   37 
   38 instance Functor Node where
   39   f `fmap` n = NodeT {key = key n, unNode = f (unNode n)}
   40 
   41 instance (Show v) => Show (Node v) where
   42   show (NodeT h v) = "NodeT { " ++ show h ++ " " ++ show v ++ " }"

   44 instance (Show v) => Show (Graph v) where
   45   show g = "GraphT Directed { unHashMap = " ++ show (unHashMap g) ++ " }"

   47 instance Eq v => Eq (Node v) where
   48   (NodeT k1 v1) == (NodeT k2 v2) = k1 == k2 && v1 == v2

   50 -- | The empty graph
   51 empty :: Graph v
   52 empty = GraphT HM.empty
   53 
   54 -- | Returns true if a node is an element of a given graph
   55 isElement :: Graph v -> Node v -> Bool
   56 isElement g n = HM.member (key n) (unHashMap g)
   57 
   58 -- | Given a graph and node, update node if already present, otherwise insert, leaving relationships unchanged
   59 insert :: Graph v -> Node v -> Graph v
   60 insert g@(GraphT hm) n@(NodeT k _)
   61   | HM.member k hm = g
   62   | otherwise = GraphT (HM.insert k (n, HS.empty) hm)
   63 
   64 -- | Create a (directed) edge from the first node to the second one, adding one or both nodes if necessary
   65 createEdge :: Graph v -> Node v -> Node v -> Graph v
   66 createEdge g v1 v2 =
   67   case (HM.lookup (key v1) (unHashMap g), HM.lookup (key v2) (unHashMap g)) of
   68     (Just x, Just y) -> GraphT (HM.update (\z -> Just (fst z, (snd z) `HS.union` HS.singleton (key (fst y)))) (key (fst x)) (unHashMap g))
   69     (Just x, Nothing) -> GraphT (HM.update (\z -> Just (fst x, (snd z) `HS.union` HS.singleton (key v2))) (key (fst x)) (HM.insert (key v2) (v2, HS.empty) (unHashMap g)))
   70     (Nothing, Just y) -> GraphT (HM.insert (key v1) (v1, HS.singleton (key (fst y))) (unHashMap g))
   71     (Nothing, Nothing) -> GraphT (HM.insert (key v1) (v1, HS.singleton (key v2)) (HM.insert (key v2) (v2, HS.empty) (unHashMap g)))
   72 
   73 -- | Return true if the first node is a child of the second
   74 adjacent :: Graph v -> Node v -> Node v -> Bool
   75 adjacent g v1 v2 = case (HM.lookup (key v1) (unHashMap g), HM.lookup (key v2) (unHashMap g)) of
   76   (Just x, Just y) -> HS.member (key (fst y)) (snd x)
   77   _ -> False
   78 
   79 parents :: Graph v -> Node v -> [Node v]
   80 parents g (NodeT k1 _) = fromMaybe [] (sequence [fst <$> HM.lookup nxt (unHashMap g) | nxt <- HS.toList (HM.findWithDefault HS.empty k1 (HM.map snd (unHashMap g)))])

   82 -- | Determines the total number of elements in a graph
   83 order :: Graph v -> Int
   84 order = HM.size . unHashMap

   86 -- | Generate a unique hash (w.r.t. a specific graph at an instant) for a value.
   87 wrapNode :: H.Hashable h => (v -> h) -> v -> Node v
   88 wrapNode f v = NodeT (H.hash (f v)) v
   89 
   90 -- | Given an adjacency list and a hash function, return a graph containing the same information
   91 withHashFunction :: H.Hashable h => (v -> h) -> [(v, [v])] -> Graph v
   92 withHashFunction f ls =
   93   let withHashFunction' _ [] acc = acc
   94       withHashFunction' h (c : cs) acc = case c of
   95         (x, []) -> withHashFunction' h cs (insert acc (wrapNode h x))
   96         (x, y : ys) -> withHashFunction' h ((x, ys) : cs) (createEdge acc (wrapNode h x) (wrapNode h y))
   97    in withHashFunction' f ls empty

   99 leastCommonAncestor :: Eq v => Graph v -> Node v -> Node v -> Maybe (Node v)
  100 leastCommonAncestor g n1 n2 =
  101   case L.intersectBy (\x1 x2 -> snd x1 == snd x2) (allAncestors 0 g n1) (allAncestors 1 g n2) of
  102     [] -> Nothing
  103     as -> Just . snd . head $ L.sortOn fst as

  105 instance (Q.Arbitrary v) => Q.Arbitrary (Node v) where
  106   arbitrary = do
  107     int <- H.hash <$> Q.vectorOf 10 (Q.arbitrary :: Q.Gen Char)
  108     val <- Q.arbitrary :: Q.Gen v
  109     return (NodeT int val)

  111 instance (Q.Arbitrary v, H.Hashable v, Eq v) => Q.Arbitrary (Graph v) where
  112   arbitrary = do
  113     ns <- Q.listOf (Q.arbitrary :: Q.Gen v)
  114     let zipped = zip (L.nub [x | x <- ns]) (take (length (L.nub ns)) (L.subsequences [x | x <- ns]))
  115     return (withHashFunction H.hash zipped)

  117 allAncestors :: Int -> Graph v -> Node v -> [(Int, Node v)]
  118 allAncestors acc g n = (acc, n) : (parents g n >>= \p -> allAncestors (acc + 1) g p)
