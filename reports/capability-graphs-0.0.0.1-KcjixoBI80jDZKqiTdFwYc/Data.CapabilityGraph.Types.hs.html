   29 -- | An individual node independent of any graph graph, including its key, additional attributes (if any), and its neighbors
   30 data Node h v = NodeT {hashKey :: h, attributes :: v, neighbors :: HM.HashMap h (Node h v)}

   40 -- | The type of a capability graph
   41 newtype CapabilityGraph h v = CapabilityGraphT
   42   { -- | returns the underlying data structure we're using.
   43     -- | Should never be used directly by non-library code
   44     unHashMap :: HM.HashMap h (Node h v)
   45   }
   46 
   47 instance (Show h, Show v) => Show (CapabilityGraph h v) where
   48   show g = "CapabilityGraphT { " ++ show (unHashMap g) ++ "}"

   50 -- | The empty graph
   51 empty :: (Eq h, H.Hashable h) => CapabilityGraph h v
   52 empty = CapabilityGraphT {unHashMap = HM.empty}

   60 nodesByValue :: Eq v => CapabilityGraph h v -> v -> [GNode h v]
   61 nodesByValue g v =
   62   map (GNodeT g) (L.filter (\n -> attributes n == v) (HM.elems (unHashMap g)))
   63 
   64 -- | Returns true if a node is an element in the graph
   65 isElement :: (Eq h, H.Hashable h) => CapabilityGraph h v -> GNode h v -> Bool
   66 isElement g (GNodeT _ n) = isJust (HM.lookup (hashKey n) (unHashMap g))
   67 
   68 -- | Determines the total number of elements in a graph
   69 order :: (Eq h, H.Hashable h) => CapabilityGraph h v -> Int
   70 order = HM.size . unHashMap
   71 
   72 -- | Insert an element into a graph (keeping the old value if present)
   73 insert :: (Eq h, H.Hashable h, Ord h, Num h) => CapabilityGraph h v -> v -> (GNode h v, CapabilityGraph h v)
   74 insert g v =
   75   let nextId = HM.foldl' (\x y -> maximum [x, hashKey y]) 0 (unHashMap g)
   76       cgraph = CapabilityGraphT (HM.union (HM.singleton nextId (NodeT nextId v HM.empty)) (unHashMap g))
   77    in (GNodeT cgraph (NodeT nextId v (maybe HM.empty neighbors (HM.lookup nextId (unHashMap cgraph)))), cgraph)

   79 instance (Q.Arbitrary h, Q.Arbitrary v, H.Hashable h, Eq h) => Q.Arbitrary (Node h v) where
   80   arbitrary = do
   81     initNode <- NodeT <$> (Q.arbitrary :: Q.Gen h) <*> (Q.arbitrary :: Q.Gen v) <*> return HM.empty
   82     hashes <- Q.listOf (Q.arbitrary :: Q.Gen h) `Q.suchThat` \x -> length x <= 10
   83     nodes <- Q.vectorOf (length hashes) (Q.arbitrary :: Q.Gen v)
   84     let parents = mconcat [HM.singleton h (NodeT h n HM.empty) | h <- hashes, n <- nodes]
   85     return (NodeT (hashKey initNode) (attributes initNode) parents)

   87 instance (Q.Arbitrary h, Q.Arbitrary v, H.Hashable h, Eq h) => Q.Arbitrary (CapabilityGraph h v) where
   88   arbitrary = do
   89     allNodes <- Q.listOf (Q.arbitrary :: Q.Gen (Node h v)) `Q.suchThat` \x -> length x <= 10
   90     let nubbedNodes = L.nubBy (\a b -> hashKey a == hashKey b) (concat [an : HM.elems (neighbors an) | an <- allNodes])
   91     return (CapabilityGraphT (foldl' (\c nxt -> HM.union c (HM.singleton (hashKey nxt) nxt)) HM.empty nubbedNodes))
