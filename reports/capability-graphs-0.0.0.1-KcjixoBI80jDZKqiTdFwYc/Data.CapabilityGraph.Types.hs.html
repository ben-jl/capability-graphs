   32 data Node v = NodeT {key :: Int, unNode :: v}
   33 
   34 instance Functor Node where
   35   f `fmap` n = NodeT {key = key n, unNode = f (unNode n)}
   36 
   37 instance (Show v) => Show (Node v) where
   38   show (NodeT h v) = "NodeT { " ++ show h ++ " " ++ show v ++ " }"

   40 -- | The type of a capability graph
   41 newtype GraphM a v = GraphMT
   42   { -- | returns the underlying data structure we're using.
   43     -- | Should never be used directly by non-library code
   44     unHashMap :: HM.HashMap Int (v, HS.HashSet Int)
   45   }
   46 
   47 instance Functor (GraphM a) where
   48   f `fmap` g = GraphMT (HM.map (BF.first f) (unHashMap g))
   49 
   50 class KeyedByInt a where
   51   keyedByInt :: a -> Int
   52 
   53 instance KeyedByInt (Node v) where
   54   keyedByInt = key
   55 
   56 instance KeyedByInt Int where
   57   keyedByInt = id
   58 
   59 instance (Show h, Show v, KeyedByInt v) => Show (GraphM h v) where
   60   show g = "GraphT { " ++ show (unHashMap g) ++ " }"
   61 
   62 -- | The empty graph
   63 empty :: KeyedByInt v => Graph v
   64 empty = GraphMT HM.empty :: GraphM () v
   65 
   66 emptyDGraph :: KeyedByInt v => DirectedGraph v
   67 emptyDGraph = GraphMT HM.empty :: GraphM Directed v
   68 
   69 emptyUGraph :: KeyedByInt v => UndirectedGraph v
   70 emptyUGraph = GraphMT HM.empty :: GraphM Undirected v

   72 wrap :: KeyedByInt v => GraphM a v -> v -> Node v
   73 wrap g v = NodeT {key = maximum (HM.keys (unHashMap g)) + 1, unNode = v}
