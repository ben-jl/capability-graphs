   20 -- | The primary graph structure, most likely will only be interacted w/ wrapped in a more featureful level
   21 newtype Graph v = GraphT
   22   { -- | underlying data structure, representing the graph as a hashmap from an INT -> a tuple containing a node (value)
   23     -- | a hashset of ints (w/ the ints being the keys of the nodes parent)
   24     unHashMap :: HM.HashMap Int (Node v, HS.HashSet Int)
   25   }
   26 
   27 instance Functor Graph where
   28   f `fmap` (GraphT hm) = GraphT $ HM.map (\(NodeT k n, hs) -> (NodeT k (f n), hs)) hm
   29 
   30 data Node v = NodeT {key :: Int, unNode :: v}
   31 
   32 instance Functor Node where
   33   f `fmap` n = NodeT {key = key n, unNode = f (unNode n)}
   34 
   35 instance (Show v) => Show (Node v) where
   36   show (NodeT h v) = "NodeT { " ++ show h ++ " " ++ show v ++ " }"
   37 
   38 instance (Show v) => Show (Graph v) where
   39   show g = "GraphT Directed { unHashMap = " ++ show (unHashMap g) ++ " }"

   41 -- | The empty graph
   42 empty :: Graph v
   43 empty = GraphT HM.empty
   44 
   45 -- | Returns true if a node is an element of a given graph
   46 isElement :: Graph v -> Node v -> Bool
   47 isElement g n = HM.member (key n) (unHashMap g)
   48 
   49 -- | Given a graph and node, update node if already present, otherwise insert, leaving relationships unchanged
   50 insert :: Graph v -> Node v -> Graph v
   51 insert g@(GraphT hm) n@(NodeT k _)
   52   | HM.member k hm = g
   53   | otherwise = GraphT (HM.insert k (n, HS.empty) hm)
   54 
   55 createEdge :: Graph v -> Node v -> Node v -> Graph v
   56 createEdge g v1 v2 =
   57   case (HM.lookup (key v1) (unHashMap g), HM.lookup (key v2) (unHashMap g)) of
   58     (Just x, Just y) -> GraphT (HM.update (\z -> Just (fst z, (snd z) <> HS.singleton (key (fst y)))) (key (fst x)) (unHashMap g))
   59     (Just x, Nothing) -> GraphT (HM.update (\z -> Just (fst x, (snd z) <> HS.singleton (key v1))) (key (fst x)) (unHashMap g))
   60     (Nothing, Just y) -> GraphT (HM.insert (key v1) (v1, HS.singleton (key (fst y))) (unHashMap g))
   61     (Nothing, Nothing) -> GraphT ((HM.insert (key v2) (v2, HS.empty)) (unHashMap g) <> HM.insert (key v1) (v2, HS.singleton (key v1)) (unHashMap g))
   62 
   63 adjacent :: Graph v -> Node v -> Node v -> Bool
   64 adjacent g v1 v2 = case (HM.lookup (key v1) (unHashMap g), HM.lookup (key v2) (unHashMap g)) of
   65   (Just x, Just y) -> HS.member (key (fst y)) (snd x)
   66   _ -> False
   67 
   68 -- | Determines the total number of elements in a graph
   69 order :: Graph v -> Int
   70 order = HM.size . unHashMap
   71 
   72 -- | Generate a unique hash (w.r.t. a specific graph at an instant) for a value.
   73 wrapContext :: Graph v -> v -> Node v
   74 wrapContext graph v =
   75   let currHashes = map (key . fst) (HM.elems (unHashMap graph))
   76       nextHash = case currHashes of
   77         [] -> 1
   78         hs -> 1 + maximum hs
   79    in NodeT nextHash v
   80 
   81 -- Given a list of tuples representing a mapping from value -> parent values (w/ the ints the values' keys), return a graph containg the same information
   82 fromKeyed :: [((Int, v), [(Int, v)])] -> Graph v
   83 fromKeyed keyed = GraphT (HM.fromList $ [(fst k1, (NodeT (fst k1) (snd k1), HS.fromList (map fst v))) | (k1, v) <- keyed] ++ [(fst k1, (NodeT (fst k1) (snd k1), HS.empty)) | k1 <- concatMap snd keyed])
