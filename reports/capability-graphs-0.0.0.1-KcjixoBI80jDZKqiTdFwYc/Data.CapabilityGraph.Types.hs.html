   16 --  (HM.HashMap h (Node v, HS.HashSet h))
   17 newtype Graph v = GraphT
   18   { unHashMap :: HM.HashMap Int (Node v, HS.HashSet Int)
   19   }
   20 
   21 data Node v = NodeT {key :: Int, unNode :: v}
   22 
   23 instance Functor Node where
   24   f `fmap` n = NodeT {key = key n, unNode = f (unNode n)}
   25 
   26 instance (Show v) => Show (Node v) where
   27   show (NodeT h v) = "NodeT { " ++ show h ++ " " ++ show v ++ " }"
   28 
   29 instance (Show v) => Show (Graph v) where
   30   show g = "GraphT Directed { unHashMap = " ++ show (unHashMap g) ++ " }"

   32 -- | The empty graph
   33 empty :: Graph v
   34 empty = GraphT HM.empty
   35 
   36 -- | Returns true if a node is an element of a given graph
   37 isElement :: Graph v -> Node v -> Bool
   38 isElement g n = HM.member (key n) (unHashMap g)
   39 
   40 -- | Given a graph and node, update node if already present, otherwise insert, leaving relationships unchanged
   41 upsert :: Graph v -> Node v -> Graph v
   42 upsert g n =
   43   let currhs = maybe HS.empty snd (HM.lookup (key n) (unHashMap g))
   44    in GraphT (HM.insert (key n) (n, currhs) (unHashMap g))
   45 
   46 -- | Determines the total number of elements in a graph
   47 order :: Graph v -> Int
   48 order = HM.size . unHashMap
