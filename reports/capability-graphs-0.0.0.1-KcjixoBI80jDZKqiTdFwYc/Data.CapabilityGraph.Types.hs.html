   28 type Graph v = (GraphM () v)

   30 type DirectedGraph v = (GraphM Directed v)

   34 data Node v = NodeT {key :: Int, unNode :: v}
   35 
   36 instance Functor Node where
   37   f `fmap` n = NodeT {key = key n, unNode = f (unNode n)}
   38 
   39 instance (Show v) => Show (Node v) where
   40   show (NodeT h v) = "NodeT { " ++ show h ++ " " ++ show v ++ " }"

   42 -- | The type of a capability graph
   43 newtype GraphM a v = GraphMT
   44   { -- | returns the underlying data structure we're using.
   45     -- | Should never be used directly by non-library code
   46     unHashMap :: HM.HashMap Int (Node v, HS.HashSet Int)
   47   }
   48 
   49 instance Functor (GraphM a) where
   50   f `fmap` g = GraphMT (HM.map (\(NodeT k v, hs) -> (NodeT k (f v), hs)) (unHashMap g))
   51 
   52 instance (Show h, Show v) => Show (GraphM h v) where
   53   show g = "GraphT { " ++ show (unHashMap g) ++ " }"
   54 
   55 -- | The empty graph
   56 empty :: Graph v
   57 empty = GraphMT HM.empty :: GraphM () v
   58 
   59 emptyDGraph :: DirectedGraph v
   60 emptyDGraph = GraphMT HM.empty :: GraphM Directed v
   61 
   62 emptyUGraph :: UndirectedGraph v
   63 emptyUGraph = GraphMT HM.empty :: GraphM Undirected v
   64 
   65 wrap :: GraphM a v -> v -> Node v
   66 wrap g v = NodeT {key = maximum (HM.keys (unHashMap g)) + 1, unNode = v}
   67 
   68 -- | Returns true if a node is an element of a given graph
   69 isElement :: GraphM a v -> Node v -> Bool
   70 isElement g n = HM.member (key n) (unHashMap g)
   71 
   72 -- | Given a graph and node, update node if already present, otherwise insert, leaving relationships unchanged
   73 upsert :: GraphM a v -> Node v -> GraphM a v
   74 upsert g n =
   75   let currhs = maybe HS.empty snd (HM.lookup (key n) (unHashMap g))
   76    in GraphMT (HM.insert (key n) (n, currhs) (unHashMap g))
   77 
   78 -- | Determines the total number of elements in a graph
   79 order :: GraphM a v -> Int
   80 order = HM.size . unHashMap
   81 
   82 -- | return a list of all nodes in the given graph
   83 nodes :: GraphM a v -> [Node v]
   84 nodes g = fst <$> HM.elems (unHashMap g)
   85 
   86 -- | Returns true if a keyable object
   87 instance (Q.Arbitrary v) => Q.Arbitrary (Node v) where
   88   arbitrary = do
   89     int <- H.hash <$> Q.vectorOf 10 (Q.arbitrary :: Q.Gen Char)
   90     val <- Q.arbitrary :: Q.Gen v
   91     return (NodeT int val)
   92 
   93 instance (Q.Arbitrary v) => Q.Arbitrary (GraphM a v) where
   94   arbitrary = do
   95     mnodes <- Q.listOf (Q.arbitrary :: Q.Gen (Node v))
   96     relatives <- Q.vectorOf (length mnodes) (Q.listOf (Q.elements mnodes))
   97     let zipped = zip mnodes relatives
   98     let res = map (\(b, ps) -> HM.singleton (key b) (b, mconcat [HS.singleton (key p) | p <- ps])) zipped
   99     return (GraphMT (mconcat res))
